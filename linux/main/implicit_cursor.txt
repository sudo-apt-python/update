DECLARE
  rows_updated INTEGER := 0;
BEGIN
  UPDATE your_table
  SET column1 = value1, column2 = value2
  WHERE condition;

  rows_updated := SQL%ROWCOUNT;

  DBMS_OUTPUT.PUT_LINE('Number of rows updated: ' || rows_updated);
EXCEPTION
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
END;
/
